local Players = game:GetService("Players")
local LocalizationService = game:GetService("LocalizationService")

local player = Players.LocalPlayer
local countryCode = "US"

-- Thử lấy mã quốc gia
local success, result = pcall(function()
    return LocalizationService:GetCountryRegionForPlayerAsync(player)
end)

if success and result then
    countryCode = result
end

-- Nếu là Việt Nam
if countryCode == "VN" then
    player:Kick("Kịch bản đã ngừng cập nhật. Vui lòng đợi đến khi có video mới để nhận.")
else
    player:Kick("The script has stopped updating. Please wait until a new video is released to receive it.")
end
